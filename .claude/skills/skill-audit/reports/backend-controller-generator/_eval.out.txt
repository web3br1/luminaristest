// server/src/controllers/invoiceController.ts
import type { Request, Response } from 'express';
import { z } from 'zod';
import { getFactory } from '@/lib/factory';
import { handleApiError } from '@/lib/apiUtils';
import { getUserContextFromRequest } from '@/lib/authUtils';

const ListInvoicesQuerySchema = z.object({
  page: z.string().optional().transform(v => (v ? parseInt(v, 10) : 1)),
  limit: z.string().optional().transform(v => (v ? parseInt(v, 10) : 10)),
  status: z.string().min(1).optional(),
});

const CreateInvoiceSchema = z.object({
  number: z.string().min(1),
  amount: z.number(),
  status: z.string().min(1),
});

const UpdateInvoiceSchema = z.object({
  number: z.string().min(1).optional(),
  amount: z.number().optional(),
  status: z.string().min(1).optional(),
});

export async function listInvoices(req: Request, res: Response) {
  try {
    const ctx = getUserContextFromRequest(req);
    if (!ctx) return res.status(401).json({ success: false, error: 'Authentication required' });

    const { page, limit, status } = ListInvoicesQuerySchema.parse(req.query);
    const svc = getFactory().getInvoiceService();

    const { invoices, totalCount } = await svc.getAllInvoices(ctx, page, limit, status);
    return res.status(200).json({ success: true, data: invoices, totalCount });
  } catch (error) {
    return handleApiError(error, res);
  }
}

export async function createInvoice(req: Request, res: Response) {
  try {
    const bodyResult = CreateInvoiceSchema.safeParse(req.body);
    if (!bodyResult.success) return res.status(400).json({ success: false, error: bodyResult.error.flatten() });

    const ctx = getUserContextFromRequest(req);
    if (!ctx) return res.status(401).json({ success: false, error: 'Authentication required' });

    const svc = getFactory().getInvoiceService();
    const created = await svc.createInvoice(bodyResult.data, ctx);
    return res.status(201).json({ success: true, data: created });
  } catch (error) {
    return handleApiError(error, res);
  }
}

export async function updateInvoice(req: Request, res: Response) {
  try {
    const bodyResult = UpdateInvoiceSchema.safeParse(req.body);
    if (!bodyResult.success) return res.status(400).json({ success: false, error: bodyResult.error.flatten() });

    const ctx = getUserContextFromRequest(req);
    if (!ctx) return res.status(401).json({ success: false, error: 'Authentication required' });

    const { id } = req.params;
    if (!id) return res.status(400).json({ success: false, error: 'Invoice ID required' });

    const svc = getFactory().getInvoiceService();
    const updated = await svc.updateInvoice(id, bodyResult.data, ctx);
    return res.status(200).json({ success: true, data: updated });
  } catch (error) {
    return handleApiError(error, res);
  }
}
