// server/src/jobs/PurgeExpiredProposals.ts
/**
 * PurgeExpiredProposals — maintenance job.
 *
 * Hard-deletes ActionProposal rows with status EXPIRED that are older than 30 days.
 *
 * Uses Prisma directly (default import from ../lib/prisma) — jobs do NOT go through
 * getFactory(). This BYPASSES the factory, DTO/schema-validation, the rule engine
 * (rules/plugins) and policy checks; acceptable for a maintenance job, but explicit.
 *
 * Idempotent: a deleteMany over `status=EXPIRED AND createdAt < cutoff` is reentrant —
 * re-running purges only what still matches and never duplicates side effects.
 *
 * Scheduling: this is a PERIODIC job. Register it at boot in `server/src/server.ts`, e.g.:
 *
 *     import { purgeExpiredProposals } from './jobs/PurgeExpiredProposals';
 *     // run daily (every 24h); fire once shortly after boot, then on an interval
 *     setInterval(() => { void purgeExpiredProposals(); }, 24 * 60 * 60 * 1000);
 *     void purgeExpiredProposals();
 */

import prisma from '../lib/prisma';
import { logger } from '../lib/logger';

const THIRTY_DAYS_MS = 30 * 24 * 60 * 60 * 1000;

export async function purgeExpiredProposals(): Promise<void> {
  const job = 'PurgeExpiredProposals';
  const cutoffDate = new Date(Date.now() - THIRTY_DAYS_MS);

  logger.info('Job started', { job, cutoff: cutoffDate.toISOString() });

  try {
    const result = await prisma.actionProposal.deleteMany({
      where: {
        status: 'EXPIRED',
        createdAt: { lt: cutoffDate },
      },
    });

    logger.info('Job completed', { job, affected: result.count });
  } catch (err) {
    logger.error('Job failed', { job, error: err instanceof Error ? err.message : String(err) });
    throw err;
  }
}
