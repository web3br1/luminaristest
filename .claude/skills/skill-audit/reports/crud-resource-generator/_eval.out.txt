import { z } from 'zod';

/**
 * Schema for comment response
 * @openapi
 * components:
 *   schemas:
 *     Comment:
 *       type: object
 *       required: [id, body, authorId, createdAt, updatedAt]
 *       properties:
 *         id: { type: string, format: cuid }
 *         body: { type: string, minLength: 1, maxLength: 5000 }
 *         authorId: { type: string, format: cuid }
 *         createdAt: { type: string, format: date-time }
 *         updatedAt: { type: string, format: date-time }
 */
export const CommentSchema = z.object({
  id: z.string().cuid({ message: 'comment.validation.idInvalidCuid' }),
  body: z.string().min(1, 'Body is required').max(5000, 'Body cannot exceed 5000 characters'),
  authorId: z.string().cuid({ message: 'comment.validation.authorIdInvalidCuid' }),
  createdAt: z.date(),
  updatedAt: z.date(),
});

/**
 * Schema for creating a comment
 * @openapi
 * components:
 *   schemas:
 *     CreateComment:
 *       type: object
 *       required: [body, authorId]
 *       properties:
 *         body: { type: string, minLength: 1, maxLength: 5000 }
 *         authorId: { type: string, format: cuid }
 */
export const CreateCommentSchema = z.object({
  body: z.string().min(1, 'Body is required').max(5000, 'Body cannot exceed 5000 characters'),
  authorId: z.string().cuid({ message: 'comment.validation.authorIdInvalidCuid' }),
});

/**
 * Schema for updating a comment
 * @openapi
 * components:
 *   schemas:
 *     UpdateComment:
 *       type: object
 *       properties:
 *         body: { type: string, minLength: 1, maxLength: 5000 }
 *         authorId: { type: string, format: cuid }
 */
export const UpdateCommentSchema = CreateCommentSchema.partial();

export type CommentDto = z.infer<typeof CommentSchema>;
export type CreateCommentInput = z.infer<typeof CreateCommentSchema>;
export type UpdateCommentInput = z.infer<typeof UpdateCommentSchema>;

// server/src/features/comments/repositories/CommentRepository.ts
import prisma from '../../../lib/prisma';
import type { Prisma, Comment } from 'generated/prisma';
import type { ICommentRepository } from './ICommentRepository';

export type CommentRecord = Comment;

/**
 * Repository implementation for Comment data access operations.
 * Every find filters out soft-deleted rows; delete is a soft-delete.
 */
export class CommentRepository implements ICommentRepository {
  /**
   * Creates a new comment.
   */
  public async create(data: Prisma.CommentCreateInput): Promise<CommentRecord> {
    return prisma.comment.create({ data });
  }

  /**
   * Retrieves a paginated list of comments.
   */
  public async findAll(page: number = 1, limit: number = 10): Promise<{ comments: CommentRecord[]; totalCount: number }> {
    const skip = (page - 1) * limit;

    const [comments, totalCount] = await prisma.$transaction([
      prisma.comment.findMany({
        where: { deletedAt: null },
        skip,
        take: limit,
        orderBy: { createdAt: 'desc' },
      }),
      prisma.comment.count({ where: { deletedAt: null } }),
    ]);

    return { comments, totalCount };
  }

  /**
   * Retrieves a comment by ID.
   */
  public async findById(id: string): Promise<CommentRecord | null> {
    return prisma.comment.findFirst({
      where: { id, deletedAt: null },
    });
  }

  /**
   * Updates a comment.
   */
  public async update(id: string, data: Prisma.CommentUpdateInput): Promise<CommentRecord> {
    return prisma.comment.update({
      where: { id },
      data,
    });
  }

  /**
   * Soft-deletes a comment.
   */
  public async softDelete(id: string): Promise<void> {
    await prisma.comment.update({
      where: { id },
      data: { deletedAt: new Date() },
    });
  }
}

import type { IUser } from '../../users/models/User.model';
import { Role } from '../../users/models/User.model';
import type { ICommentPolicy } from './ICommentPolicy';

/**
 * Authorization rules for Comment actions.
 * ADMIN can do everything; USER only what is theirs (ownership by authorId).
 */
export class CommentPolicy implements ICommentPolicy {
  canListAll(actor: IUser | null): boolean {
    return !!actor;
  }

  canView(actor: IUser | null, ownerId: string): boolean {
    if (!actor) return false;
    if (actor.role === Role.ADMIN) return true;
    return actor.id === ownerId;
  }

  canCreate(actor: IUser | null): boolean {
    return !!actor;
  }

  canUpdate(actor: IUser | null, ownerId: string): boolean {
    if (!actor) return false;
    if (actor.role === Role.ADMIN) return true;
    return actor.id === ownerId;
  }

  canDelete(actor: IUser | null, ownerId: string): boolean {
    if (!actor) return false;
    if (actor.role === Role.ADMIN) return true;
    return actor.id === ownerId;
  }
}

// server/src/features/comments/services/CommentService.ts
import type { ICommentRepository } from '../repositories/ICommentRepository';
import type { CommentRecord } from '../repositories/CommentRepository';
import type { ICommentPolicy } from '../policies/ICommentPolicy';
import type { IUser } from '../../users/models/User.model';
import { Role } from '../../users/models/User.model';
import { CreateCommentInput, UpdateCommentInput, isCreateCommentInput, isUpdateCommentInput } from '../dtos/CommentDto';
import { ForbiddenError, NotFoundError, UnauthorizedError, ValidationError } from '../../../lib/errors';

export class CommentService {
  constructor(
    private commentRepository: ICommentRepository,
    private commentPolicy: ICommentPolicy
  ) { }

  /**
   * Creates a comment.
   * @throws {UnauthorizedError} If not authenticated
   * @throws {ForbiddenError} If actor is not authorized
   * @throws {ValidationError} If data is invalid
   */
  public async createComment(data: CreateCommentInput, actor: IUser | null): Promise<CommentRecord> {
    if (!actor) {
      throw new UnauthorizedError('Authentication required to create a comment');
    }
    if (!this.commentPolicy.canCreate(actor)) {
      throw new ForbiddenError('You are not authorized to create comments');
    }
    if (!isCreateCommentInput(data)) {
      throw new ValidationError('Invalid comment creation data');
    }

    const authorId = actor.role === Role.ADMIN ? data.authorId : actor.id;

    return this.commentRepository.create({
      body: data.body,
      author: { connect: { id: authorId } },
    });
  }

  /**
   * Retrieves a paginated list of comments.
   * @throws {ForbiddenError} If actor is not authorized
   */
  public async getAllComments(actor: IUser | null, page?: number, limit?: number): Promise<{ comments: CommentRecord[]; totalCount: number }> {
    if (!this.commentPolicy.canListAll(actor)) {
      throw new ForbiddenError('You are not authorized to list comments');
    }
    return this.commentRepository.findAll(page, limit);
  }

  /**
   * Retrieves a comment by ID.
   * @throws {NotFoundError} If not found or not visible to the actor
   */
  public async getCommentById(id: string, actor: IUser | null): Promise<CommentRecord> {
    const comment = await this.commentRepository.findById(id);
    if (!comment) {
      throw new NotFoundError('Comment not found');
    }
    if (!this.commentPolicy.canView(actor, comment.authorId)) {
      throw new NotFoundError('Comment not found');
    }
    return comment;
  }

  /**
   * Updates a comment.
   * @throws {ForbiddenError} If actor is not authorized
   * @throws {NotFoundError} If not found
   */
  public async updateComment(id: string, data: UpdateCommentInput, actor: IUser | null): Promise<CommentRecord> {
    if (!actor) {
      throw new UnauthorizedError('Authentication required to update a comment');
    }
    if (!isUpdateCommentInput(data)) {
      throw new ValidationError('Invalid comment update data');
    }

    const existing = await this.commentRepository.findById(id);
    if (!existing) {
      throw new NotFoundError('Comment not found');
    }
    if (!this.commentPolicy.canUpdate(actor, existing.authorId)) {
      throw new ForbiddenError('You are not authorized to update this comment');
    }
    if (data.body === undefined) {
      throw new ValidationError('No valid update fields provided');
    }

    return this.commentRepository.update(id, { body: data.body });
  }

  /**
   * Soft-deletes a comment.
   * @throws {ForbiddenError} If actor is not authorized
   * @throws {NotFoundError} If not found
   */
  public async deleteComment(id: string, actor: IUser | null): Promise<void> {
    if (!actor) {
      throw new UnauthorizedError('Authentication required to delete a comment');
    }

    const existing = await this.commentRepository.findById(id);
    if (!existing) {
      throw new NotFoundError('Comment not found');
    }
    if (!this.commentPolicy.canDelete(actor, existing.authorId)) {
      throw new ForbiddenError('You are not authorized to delete this comment');
    }

    await this.commentRepository.softDelete(id);
  }
}

    const page = parseInt(req.query.page as string) || 1;

import { apiClient } from '../api/api-client';

export interface Comment {
  id: string;
  body: string;
  authorId: string;
  createdAt: string;
  updatedAt: string;
}

export interface CreateCommentPayload {
  body: string;
  authorId: string;
}

export type UpdateCommentPayload = Partial<CreateCommentPayload>;

interface PaginationMetadata {
  page: number;
  limit: number;
  totalCount: number;
  totalPages: number;
}

/**
 * Comment Service (Client-side).
 */
export const CommentService = {
  /**
   * Fetches a paginated list of comments.
   */
  async getComments(page: number, limit: number): Promise<{ data: Comment[]; pagination: PaginationMetadata }> {
    return apiClient.get<{ data: Comment[]; pagination: PaginationMetadata }>(`/comments?page=${page}&limit=${limit}`);
  },

  /**
   * Fetches a single comment by ID.
   */
  async getCommentById(id: string): Promise<Comment> {
    const response = await apiClient.get<{ data: Comment }>(`/comments/${id}`);
    return response.data;
  },

  /**
   * Creates a comment.
   */
  async createComment(payload: CreateCommentPayload): Promise<Comment> {
    const response = await apiClient.post<{ data: Comment }>('/comments', payload);
    return response.data;
  },

  /**
   * Updates a comment.
   */
  async updateComment(id: string, payload: UpdateCommentPayload): Promise<Comment> {
    const response = await apiClient.put<{ data: Comment }>(`/comments/${id}`, payload);
    return response.data;
  },

  /**
   * Deletes a comment.
   */
  async deleteComment(id: string): Promise<void> {
    await apiClient.delete(`/comments/${id}`);
  },
};

===regression-1===
Tecnologia correta: **prisma** (first-class).

Pelo teste do `_ARCHITECTURE-CONTRACT.md §2.1`:

1. `amount` é dinheiro que entra no saldo do cliente — invariante financeiro. Em DynamicTable o valor viveria em `data: Json`, que no SQLite é float; centavos exigem inteiro real (Int).
2. Saldo/fechamento exige integridade garantida pelo banco (`@@unique`, FK, tipos reais). O `unique()` de preset é scan em nível de aplicação (TOCTOU), não constraint.
3. Invoice não é schema que o usuário configura em runtime — é entidade fixa do ERP com regra contábil.

Logo: Model Prisma `Invoice` + `InvoiceRepository` + `InvoicePolicy` + `InvoiceService` próprios, `amount` em centavos (Int), soft-delete (`deletedAt: null` em todo find, delete via `update({ data: { deletedAt } })`).

Onde a integração com `PostingService` deve viver:

- **Nunca** dentro do motor de plugins: não injete `PostingService` em `DynamicTableService`, `RuleContext` ou `RulePlugin`, e não modifique `DynamicTableService.ts` para integrar os dois módulos. É a fronteira dura do §2.1.
- A integração cross-módulo sobe de nível: quem chama `PostingService` é a camada de orquestração da própria feature — o `InvoiceService` (ou um serviço de integração dedicado) acionado a partir do controller/rota de invoices, depois de persistir a Invoice.
- O gate autoritativo (período aberto, saldo, status) é re-checado **dentro** do `runTransaction`, com o `tx` propagado ao repositório, para que a escrita da Invoice e o lançamento contábil sejam atômicos — `@@unique` sozinho não fecha o TOCTOU.
- A rota nasce protegida (deny-by-default); registro em 2 toques: mount em `routes/index.ts` + bloco `@openapi` em `routes/docs.paths.ts`. Nada em `middleware/auth.ts`.
