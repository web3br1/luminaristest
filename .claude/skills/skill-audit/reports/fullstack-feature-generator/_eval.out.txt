 *   schemas:
 *     Invoice:
 *       type: object
 *       required: [id, userId, unitId, customerName, issueDate, dueDate, amountCents, status, createdAt, updatedAt]
 *       properties:
 *         id: { type: string, format: cuid }
 *         userId: { type: string, format: cuid }
 *         unitId: { type: string }
 *         customerName: { type: string, minLength: 1, maxLength: 200 }
 *         documentNumber: { type: string, nullable: true, maxLength: 60 }
 *         description: { type: string, maxLength: 500 }
 *         issueDate: { type: string, format: date, description: 'YYYY-MM-DD' }
 *         dueDate: { type: string, format: date, description: 'YYYY-MM-DD' }
 *         amountCents: { type: integer, description: 'Valor em centavos inteiros' }
 *         status: { type: string, enum: [OPEN, PAID, CANCELLED] }
 *         createdAt: { type: string, format: date-time }
 *         updatedAt: { type: string, format: date-time }
 *         deletedAt: { type: string, format: date-time, nullable: true }
 *     CreateInvoice:
 *       type: object
 *       required: [unitId, customerName, description, issueDate, dueDate, amountCents]
 *       properties:
 *         unitId: { type: string }
 *         customerName: { type: string, minLength: 1, maxLength: 200 }
 *         documentNumber: { type: string, maxLength: 60 }
 *         description: { type: string, maxLength: 500 }
 *         issueDate: { type: string, format: date }
 *         dueDate: { type: string, format: date }
 *         amountCents: { type: integer, minimum: 1 }
 *     UpdateInvoice:
 *       type: object
 *       description: Todos os campos de CreateInvoice, opcionais (partial)
 */

export const InvoiceStatusSchema = z.enum(['OPEN', 'PAID', 'CANCELLED']);
export type InvoiceStatus = z.infer<typeof InvoiceStatusSchema>;

/** `YYYY-MM-DD` com validação de calendário (round-trip), não só regex. */
const DateOnlySchema = z
  .string()
  .regex(/^\d{4}-\d{2}-\d{2}$/, 'invoice.validation.dateOnlyFormat')
  .refine((value) => {
    const parsed = new Date(`${value}T00:00:00.000Z`);
    return !Number.isNaN(parsed.getTime()) && parsed.toISOString().slice(0, 10) === value;
  }, 'invoice.validation.dateOnlyCalendar');

export const CreateInvoiceSchema = z.object({
  unitId: z.string().min(1, 'invoice.validation.unitIdRequired'),
  customerName: z
    .string()
    .min(1, 'invoice.validation.customerNameRequired')
    .max(200, 'invoice.validation.customerNameTooLong'),
  documentNumber: z.string().max(60, 'invoice.validation.documentNumberTooLong').optional(),
  description: z.string().max(500, 'invoice.validation.descriptionTooLong'),
  issueDate: DateOnlySchema,
  dueDate: DateOnlySchema,
  amountCents: z
    .number()
    .int('invoice.validation.amountCentsMustBeInteger')
    .positive('invoice.validation.amountCentsMustBePositive'),
});

export const UpdateInvoiceSchema = CreateInvoiceSchema.partial();

export const InvoiceSchema = z.object({
  id: z.string().cuid({ message: 'invoice.validation.idInvalidCuid' }),
  userId: z.string().cuid(),
  unitId: z.string(),
  customerName: z.string(),
  documentNumber: z.string().nullable(),
  description: z.string(),
  issueDate: DateOnlySchema,
  dueDate: DateOnlySchema,
  amountCents: z.number().int(),
  status: InvoiceStatusSchema,
  createdAt: z.date(),
  updatedAt: z.date(),
  deletedAt: z.date().nullable(),
});

export type InvoiceDto = z.infer<typeof InvoiceSchema>;
export type CreateInvoiceDto = z.infer<typeof CreateInvoiceSchema>;
export type UpdateInvoiceDto = z.infer<typeof UpdateInvoiceSchema>;

/** Type guard para CreateInvoiceDto. */
export function isCreateInvoiceInput(obj: unknown): obj is CreateInvoiceDto {
  return CreateInvoiceSchema.safeParse(obj).success;
}

/** Type guard para UpdateInvoiceDto. */
export function isUpdateInvoiceInput(obj: unknown): obj is UpdateInvoiceDto {
  return UpdateInvoiceSchema.safeParse(obj).success;
}

import type { IInvoiceRepository } from '../repositories/IInvoiceRepository';
import type { IInvoicePolicy } from '../policies/IInvoicePolicy';
import type { IUser } from '../../users/models/User.model';
import type { IInvoice } from '../models/Invoice.model';
import {
  CreateInvoiceDto,
  UpdateInvoiceDto,
  isCreateInvoiceInput,
  isUpdateInvoiceInput,
} from '../dtos/InvoiceDto';
import { ForbiddenError, NotFoundError, ValidationError } from '../../../lib/errors';

export interface ListInvoicesResult {
  invoices: IInvoice[];
  totalCount: number;
}

export class InvoiceService {
  constructor(
    private readonly invoiceRepository: IInvoiceRepository,
    private readonly invoicePolicy: IInvoicePolicy,
  ) {}

  /**
   * Lista faturas do tenant, paginado.
   * @throws {ForbiddenError} Se a policy negar.
   */
  public async listInvoices(
    actor: IUser | null,
    unitId: string,
    page = 1,
    limit = 10,
  ): Promise<ListInvoicesResult> {
    if (!this.invoicePolicy.canList(actor)) {
      throw new ForbiddenError('You are not authorized to list invoices');
    }

    return this.invoiceRepository.findAll({ userId: actor!.id, unitId, page, limit });
  }

  /**
   * Busca uma fatura por id dentro do tenant do actor.
   * @throws {ForbiddenError} Se a policy negar.
   * @throws {NotFoundError} Se não existir, estiver soft-deleted ou for de outro tenant.
   */
  public async getInvoiceById(id: string, actor: IUser | null): Promise<IInvoice> {
    if (!this.invoicePolicy.canView(actor)) {
      throw new ForbiddenError('You are not authorized to view this invoice');
    }

    const invoice = await this.invoiceRepository.findById(id);
    if (!invoice || invoice.userId !== actor!.id) {
      throw new NotFoundError('Invoice not found');
    }

    return invoice;
  }

  /**
   * Cria uma fatura.
   * @throws {ForbiddenError} Se a policy negar.
   * @throws {ValidationError} Se o payload não bater com o DTO.
   */
  public async createInvoice(data: CreateInvoiceDto, actor: IUser | null): Promise<IInvoice> {
    if (!this.invoicePolicy.canCreate(actor)) {
      throw new ForbiddenError('You are not authorized to create invoices');
    }
    if (!isCreateInvoiceInput(data)) {
      throw new ValidationError('Invalid invoice creation data');
    }

    return this.invoiceRepository.create({ ...data, userId: actor!.id, status: 'OPEN' });
  }

  /**
   * Atualiza uma fatura existente.
   * @throws {ForbiddenError} Se a policy negar.
   * @throws {ValidationError} Se o payload não bater com o DTO.
   * @throws {NotFoundError} Se não existir ou for de outro tenant.
   */
  public async updateInvoice(
    id: string,
    data: UpdateInvoiceDto,
    actor: IUser | null,
  ): Promise<IInvoice> {
    if (!this.invoicePolicy.canUpdate(actor)) {
      throw new ForbiddenError('You are not authorized to update this invoice');
    }
    if (!isUpdateInvoiceInput(data)) {
      throw new ValidationError('Invalid invoice update data');
    }

    const existing = await this.invoiceRepository.findById(id);
    if (!existing || existing.userId !== actor!.id) {
      throw new NotFoundError('Invoice not found');
    }

    return this.invoiceRepository.update(id, data);
  }

  /**
   * Remove logicamente uma fatura (soft-delete no repositório).
   * @throws {ForbiddenError} Se a policy negar.
   * @throws {NotFoundError} Se não existir ou for de outro tenant.
   */
  public async deleteInvoice(id: string, actor: IUser | null): Promise<void> {
    if (!this.invoicePolicy.canDelete(actor)) {
      throw new ForbiddenError('You are not authorized to delete this invoice');
    }

    const existing = await this.invoiceRepository.findById(id);
    if (!existing || existing.userId !== actor!.id) {
      throw new NotFoundError('Invoice not found');
    }

    await this.invoiceRepository.softDelete(id);
  }
}

import { Request, Response } from 'express';
import { handleApiError } from '../lib/apiUtils';
import { getFactory } from '@/lib/factory';

import { CreateInvoiceSchema, UpdateInvoiceSchema } from '@/features/invoices/dtos/InvoiceDto';

export const getInvoices = async (req: Request, res: Response) => {
  try {
    const unitId = String(req.query.unitId ?? '');
    if (!unitId) {
      return res.status(400).json({ success: false, error: 'unitId is required' });
    }
    const page = parseInt(req.query.page as string) || 1;
    const limit = parseInt(req.query.limit as string) || 10;

    const actor = getUserContextFromRequest(req);
    const service = getFactory().getInvoiceService();
    const { invoices, totalCount } = await service.listInvoices(actor, unitId, page, limit);

    return res.json({
      success: true,
      data: invoices,
      pagination: {
        page,
        limit,
        totalCount,
        totalPages: Math.ceil(totalCount / limit),
      },
    });
  } catch (error) {
    return handleApiError(error, res);
  }
};

export const getInvoiceById = async (req: Request, res: Response) => {
  try {
    const { id } = req.params;
    const actor = getUserContextFromRequest(req);
    const service = getFactory().getInvoiceService();
    const invoice = await service.getInvoiceById(id, actor);

    return res.json({ success: true, data: invoice });
  } catch (error) {
    return handleApiError(error, res);
  }
};

export const createInvoice = async (req: Request, res: Response) => {
  try {
    const parse = CreateInvoiceSchema.safeParse(req.body);
    if (!parse.success) {
      return res.status(400).json({ success: false, error: parse.error.flatten() });
    }

    const actor = getUserContextFromRequest(req);
    const service = getFactory().getInvoiceService();
    const created = await service.createInvoice(parse.data, actor);

    return res.status(201).json({ success: true, data: created });
  } catch (error) {
    return handleApiError(error, res);
  }
};

export const updateInvoice = async (req: Request, res: Response) => {
  try {
    const { id } = req.params;
    const parse = UpdateInvoiceSchema.safeParse(req.body);
    if (!parse.success) {
      return res.status(400).json({ success: false, error: parse.error.flatten() });
    }

    const actor = getUserContextFromRequest(req);
    const service = getFactory().getInvoiceService();
    const updated = await service.updateInvoice(id, parse.data, actor);

    return res.json({ success: true, data: updated });
  } catch (error) {
    return handleApiError(error, res);
  }

// my-app/lib/services/invoice.service.ts
import { apiClient } from '../api/api-client';

export interface PaginationMetadata {
  page: number;
  limit: number;
  totalCount: number;
  totalPages: number;
}

export type InvoiceStatus = 'OPEN' | 'PAID' | 'CANCELLED';

export interface Invoice {
  id: string;
  userId: string;
  unitId: string;
  customerName: string;
  documentNumber: string | null;
  description: string;
  /** YYYY-MM-DD */
  issueDate: string;
  /** YYYY-MM-DD */
  dueDate: string;
  /** Centavos inteiros. */
  amountCents: number;
  status: InvoiceStatus;
  createdAt: string;
  updatedAt: string;
  deletedAt: string | null;
}

export interface ListInvoicesQuery {
  unitId: string;
  page?: number;
  limit?: number;
}

export interface CreateInvoicePayload {
  unitId: string;
  customerName: string;
  documentNumber?: string;
  description: string;
  issueDate: string;
  dueDate: string;
  amountCents: number;
}

export type UpdateInvoicePayload = Partial<CreateInvoicePayload>;

function buildQuery(params: Record<string, string | undefined>): string {
  const pairs = Object.entries(params)
    .filter(([, v]) => v !== undefined && v !== '')
    .map(([k, v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v as string)}`);
  return pairs.length ? `?${pairs.join('&')}` : '';
}

export const invoiceService = {
  async listInvoices(
    query: ListInvoicesQuery,
  ): Promise<{ data: Invoice[]; pagination: PaginationMetadata }> {
    const qs = buildQuery({
      unitId: query.unitId,
      page: query.page !== undefined ? String(query.page) : undefined,
      limit: query.limit !== undefined ? String(query.limit) : undefined,
    });
    return apiClient.get<{ data: Invoice[]; pagination: PaginationMetadata }>(`/invoices${qs}`);
  },

  async getInvoiceById(id: string): Promise<Invoice> {
    const res = await apiClient.get<{ success: boolean; data: Invoice }>(
      `/invoices/${encodeURIComponent(id)}`,
    );
    return res.data;
  },

  async createInvoice(payload: CreateInvoicePayload): Promise<Invoice> {
    const res = await apiClient.post<{ success: boolean; data: Invoice }>('/invoices', payload);
    return res.data;
  },

  async updateInvoice(id: string, payload: UpdateInvoicePayload): Promise<Invoice> {
    const res = await apiClient.put<{ success: boolean; data: Invoice }>(
      `/invoices/${encodeURIComponent(id)}`,
      payload,
    );
    return res.data;
  },

  async deleteInvoice(id: string): Promise<void> {
    await apiClient.delete(`/invoices/${encodeURIComponent(id)}`);
  },
};

function toRecord(invoice: Invoice): GenericRecord {
  return {
    id: invoice.id,
    data: {
      customerName: invoice.customerName,
      documentNumber: invoice.documentNumber,
      issueDate: invoice.issueDate,
      dueDate: invoice.dueDate,
      amountCents: invoice.amountCents,
      status: invoice.status,
    },

  unitId: string;
}

function InvoicesPage({ unitId }: InvoicesPageProps) {
  const { t } = useTranslation('invoices');
  const [invoices, setInvoices] = useState<Invoice[]>([]);
  const [pagination, setPagination] = useState<PaginationMetadata | null>(null);
  const [page, setPage] = useState(1);

  const load = useCallback(async () => {
    const res = await invoiceService.listInvoices({ unitId, page, limit: PAGE_SIZE });
    setInvoices(res.data);
    setPagination(res.pagination);
  }, [unitId, page]);

  useEffect(() => {
    void load();
  }, [load]);

  const records = useMemo(() => invoices.map(toRecord), [invoices]);

  const handleDelete = useCallback(
    async (record: GenericRecord) => {
      await invoiceService.deleteInvoice(record.id);
      await load();
    },
    [load],
  );

  return (
    <div className="space-y-4 p-6">
      <header>
        <h1 className="text-2xl font-semibold text-neutral-900 dark:text-neutral-100">
          {t('invoices.title', 'Faturas')}
        </h1>
        <p className="text-sm text-neutral-500 dark:text-neutral-400">
          {t('invoices.subtitle', 'Faturas emitidas para a unidade')}
        </p>
      </header>

      <div className="rounded-2xl border border-neutral-200 bg-white dark:border-neutral-800 dark:bg-neutral-900">
        <GenericTable
          records={records}
          schema={INVOICE_SCHEMA}
          tableId="invoices"
          relationLookups={{}}
          onEditSuccess={load}
          onDeleteConfirm={handleDelete}
        />
      </div>

      {pagination && (
        <StandardPagination
          currentPage={pagination.page}
          totalPages={pagination.totalPages}
          totalItems={pagination.totalCount}
          itemsPerPage={pagination.limit}
          onPageChange={setPage}
        />
      )}
    </div>
  );
}

// server/src/features/invoices/services/InvoiceService.ts
import type { IInvoiceRepository } from '../repositories/IInvoiceRepository';
import type { IInvoicePolicy } from '../policies/IInvoicePolicy';
import type { IUser } from '../../users/models/User.model';
import type { IInvoice } from '../models/Invoice.model';
import { CreateInvoiceDto, isCreateInvoiceInput } from '../dtos/InvoiceDto';
import { ForbiddenError, NotFoundError, ValidationError } from '../../../lib/errors';

export interface ListInvoicesResult {
  invoices: IInvoice[];
  totalCount: number;
}

export class InvoiceService {
  constructor(
    private readonly invoiceRepository: IInvoiceRepository,
    private readonly invoicePolicy: IInvoicePolicy,
  ) {}

  public async listInvoices(
    actor: IUser | null,
    unitId: string,
    page = 1,
    limit = 10,
  ): Promise<ListInvoicesResult> {
    if (!this.invoicePolicy.canList(actor)) {
      throw new ForbiddenError('You are not authorized to list invoices');
    }

    // Toda leitura passa pelo IInvoiceRepository injetado; a query e o filtro de
    // soft-delete (deletedAt: null) vivem dentro da camada de repositório.
    return this.invoiceRepository.findAll({ userId: actor!.id, unitId, page, limit });
  }

  public async getInvoiceById(id: string, actor: IUser | null): Promise<IInvoice> {
    if (!this.invoicePolicy.canView(actor)) {
      throw new ForbiddenError('You are not authorized to view this invoice');
    }

    const invoice = await this.invoiceRepository.findById(id);
    if (!invoice || invoice.userId !== actor!.id) {
      throw new NotFoundError('Invoice not found');
    }

    return invoice;
  }

  public async createInvoice(data: CreateInvoiceDto, actor: IUser | null): Promise<IInvoice> {
    if (!this.invoicePolicy.canCreate(actor)) {
      throw new ForbiddenError('You are not authorized to create invoices');
    }
    if (!isCreateInvoiceInput(data)) {
      throw new ValidationError('Invalid invoice creation data');
    }

    return this.invoiceRepository.create({ ...data, userId: actor!.id, status: 'OPEN' });
  }

  public async deleteInvoice(id: string, actor: IUser | null): Promise<void> {
    if (!this.invoicePolicy.canDelete(actor)) {
      throw new ForbiddenError('You are not authorized to delete this invoice');
    }

    const existing = await this.invoiceRepository.findById(id);
    if (!existing || existing.userId !== actor!.id) {
      throw new NotFoundError('Invoice not found');
    }

    // O resultado é sinalizado pelo tipo de retorno e por erro tipado de lib/errors.
    // Quem traduz isso em status HTTP e corpo de resposta é o invoiceController,
    // via handleApiError — esta camada não conhece o transporte.
    await this.invoiceRepository.softDelete(id);
  }
}

import { Request, Response } from 'express';
import { handleApiError } from '../lib/apiUtils';
import { getFactory } from '@/lib/factory';
import { getUserContextFromRequest } from '@/lib/authUtils';

/**
 * GET /api/invoices?unitId=&page=&limit=
 *
 * Envelope de listagem: { success, data, pagination }.
 * `data[].amountCents` é INTEIRO em centavos — este é o nome de campo que o
 * frontend service (my-app/lib/services/invoice.service.ts) tipa e consome.
 */
export const getInvoices = async (req: Request, res: Response) => {
  try {
    const unitId = String(req.query.unitId ?? '');
    if (!unitId) {
      return res.status(400).json({ success: false, error: 'unitId is required' });
    }
    const page = parseInt(req.query.page as string) || 1;
    const limit = parseInt(req.query.limit as string) || 10;

    const actor = getUserContextFromRequest(req);
    const service = getFactory().getInvoiceService();
    const { invoices, totalCount } = await service.listInvoices(actor, unitId, page, limit);

    const data = invoices.map((invoice) => ({
      id: invoice.id,
      userId: invoice.userId,
      unitId: invoice.unitId,
      customerName: invoice.customerName,
      documentNumber: invoice.documentNumber,
      description: invoice.description,
      issueDate: invoice.issueDate,
      dueDate: invoice.dueDate,
      amountCents: invoice.amountCents,
      status: invoice.status,
      createdAt: invoice.createdAt,
      updatedAt: invoice.updatedAt,
      deletedAt: invoice.deletedAt,
    }));

    return res.json({
      success: true,
      data,
      pagination: {
        page,
        limit,
        totalCount,
        totalPages: Math.ceil(totalCount / limit),
      },
    });
  } catch (error) {
    return handleApiError(error, res);
  }
};

// my-app/lib/services/invoice.service.ts
import { apiClient } from '../api/api-client';

/** Espelha o objeto `pagination` que o controller emite em GET /api/invoices. */
export interface PaginationMetadata {
  page: number;
  limit: number;
  totalCount: number;
  totalPages: number;
}

export type InvoiceStatus = 'OPEN' | 'PAID' | 'CANCELLED';

/** Espelha um item de `data[]` do envelope de listagem do backend. */
export interface Invoice {
  id: string;
  userId: string;
  unitId: string;
  customerName: string;
  documentNumber: string | null;
  description: string;
  /** YYYY-MM-DD */
  issueDate: string;
  /** YYYY-MM-DD */
  dueDate: string;
  /** Mesmo nome e mesma unidade do backend: INTEIRO em centavos. */
  amountCents: number;
  status: InvoiceStatus;
  createdAt: string;
  updatedAt: string;
  deletedAt: string | null;
}

export interface ListInvoicesQuery {
  unitId: string;
  page?: number;
  limit?: number;
}

function buildQuery(params: Record<string, string | undefined>): string {
  const pairs = Object.entries(params)
    .filter(([, v]) => v !== undefined && v !== '')
    .map(([k, v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v as string)}`);
  return pairs.length ? `?${pairs.join('&')}` : '';
}

export const invoiceService = {
  /** GET /api/invoices — devolve o envelope de listagem tal como o backend o emite. */
  async listInvoices(
    query: ListInvoicesQuery,
  ): Promise<{ data: Invoice[]; pagination: PaginationMetadata }> {
    const qs = buildQuery({
      unitId: query.unitId,
      page: query.page !== undefined ? String(query.page) : undefined,
      limit: query.limit !== undefined ? String(query.limit) : undefined,
    });
    return apiClient.get<{ data: Invoice[]; pagination: PaginationMetadata }>(`/invoices${qs}`);
  },
};
