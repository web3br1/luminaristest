// server/src/features/analytics/kpis/revenue/TicketMedioKpiProcessor.ts
import type { AnalyticsProcessor, ChartDataPoint } from '../../core'
import { DataSanitizer } from '../../utils/DataSanitizer'
import { addMoney } from '../../utils/CurrencyUtils'
import { getPeriodBoundaries, isDateWithinWindow } from '../../utils/DateUtils'

export const ticketMedioKpiProcessor: AnalyticsProcessor = async (context): Promise<ChartDataPoint[]> => {
  const { rows, params, table } = context
  const stream = typeof context.streamRows === 'function' ? context.streamRows() : (async function* () { yield rows })()
  let currentTotal = 0; let currentCount = 0; let previousCount = 0; let previousTotal = 0
  for await (const batch of stream) {
    for (const row of batch) {
      const data = (row.data ?? row) as Record<string, unknown>
      const amount = DataSanitizer.extractCurrency(data['totalAmount'])
      if (!Number.isFinite(amount) || amount <= 0) continue
      currentTotal = addMoney(currentTotal, amount); currentCount++
    }
  }
  const value = currentCount > 0 ? currentTotal / currentCount : 0
  const previousValue: number | undefined = previousCount > 0 ? previousTotal / previousCount : undefined
  return [{ name: 'Ticket Médio', value, previousValue, recordIds: [], tableSource: (table as any).presetKey }]
}
