import { useMemo } from 'react'
import { useTranslation } from 'next-i18next'

  const { t } = useTranslation('common')

  const rows = useMemo(() => data ?? [], [data])

    <div className="flex h-full w-full flex-col">
      <header className="flex items-center justify-between px-6 py-4"><h1 className="text-xl font-medium text-neutral-100">{t('appointments.title', 'Agendamentos')}</h1></header>
      <div className="flex-1 overflow-auto px-6 pb-6">
        {rows.length === 0 ? <p className="text-sm text-neutral-400">{t('appointments.empty', 'Nenhum.')}</p> : <ul className="flex flex-col gap-2">{rows.map((row) => <li key={row.id} className="rounded-2xl border border-neutral-800 bg-neutral-900/40 px-4 py-3 text-sm text-neutral-200">{row.title}</li>)}</ul>}
      </div>
    </div>

}

// features/dashboard/category-views/appointments/hooks/useAppointmentsData.ts
import { useCallback, useEffect, useState } from 'react'
import { useTranslation } from 'next-i18next'
import { appointmentsService } from '../../../../../lib/services/appointments.service'

export interface AppointmentRow { id: string; title: string }
export interface UseAppointmentsDataResult { data: AppointmentRow[] | null; isLoading: boolean; error: string | null; refetch: () => Promise<void> }

export function useAppointmentsData(): UseAppointmentsDataResult {
  const { t } = useTranslation('common')
  const [data, setData] = useState<AppointmentRow[] | null>(null)
  const [isLoading, setIsLoading] = useState<boolean>(true)
  const [error, setError] = useState<string | null>(null)
  const fetchData = useCallback(async () => {
    setIsLoading(true); setError(null)
    try { const rows = await appointmentsService.list(); setData(rows) }
    catch { setError(t('appointments.error', 'Falha.')) }
    finally { setIsLoading(false) }
  }, [t])
  useEffect(() => { void fetchData() }, [fetchData])
  return { data, isLoading, error, refetch: fetchData }
}
